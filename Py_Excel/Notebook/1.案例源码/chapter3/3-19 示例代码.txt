# -*- coding: UTF-8 -*-

a = "Hello World"
print("格式化多个字符：", "%s" % a)

a = "H"
print("%c只能格式化单个字符：", "%c" % a)

a = 8
print("格式化整数：", "%d%%" % a)

a = 8.888888
print("格式化浮点数，保留2位小数：", "%.2f" % a)

a = 8.888888
print("格式化成整数：", "%+u" % a)

a = 8.888888
print("转换成科学计数法：", "%e" % a)

a = 8.888888
b = 9.999999
print("第一个数：%s，第二个数：%s" % (a, b))
